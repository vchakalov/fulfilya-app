-- PaymentSummaryReport - Daily payment analytics
WITH daily_orders AS (
  SELECT 
    DATE(o.created_at) as report_date,
    o.uuid,
    o.status,
    MAX(CASE WHEN cf.name = 'payment-method' THEN cfv.value END) as payment_method,
    MAX(CASE WHEN cf.name = 'cod-amount:' THEN REPLACE(REPLACE(cfv.value, '$', ''), ',', '') END) as cod_amount
  FROM orders o
  LEFT JOIN custom_field_values cfv ON cfv.subject_uuid = o.uuid AND cfv.subject_type = 'Fleetbase\\FleetOps\\Models\\Order'
  LEFT JOIN custom_fields cf ON cfv.custom_field_uuid = cf.uuid
  WHERE o.company_uuid = '{{appsmith.store.customer_uuid}}'
    AND o.deleted_at IS NULL
    AND DATE(o.created_at) >= DATE('{{StartDatePicker.startDate || moment().subtract(30, "days").format("YYYY-MM-DD")}}')
    AND DATE(o.created_at) <= DATE('{{StartDatePicker.endDate || moment().format("YYYY-MM-DD")}}')
  GROUP BY DATE(o.created_at), o.uuid
)
SELECT 
  report_date,
  payment_method,
  
  -- Order Counts
  COUNT(*) as total_orders,
  SUM(CASE WHEN status = 'delivered' THEN 1 ELSE 0 END) as delivered,
  SUM(CASE WHEN status = 'cancelled' THEN 1 ELSE 0 END) as cancelled,
  SUM(CASE WHEN status IN ('created', 'dispatched', 'in_transit') THEN 1 ELSE 0 END) as in_progress,
  
  -- Success Rate
  ROUND(SUM(CASE WHEN status = 'delivered' THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 1) as delivery_rate,
  
  -- COD Financials
  SUM(CASE 
    WHEN payment_method = 'COD' AND status = 'delivered' 
    THEN CAST(cod_amount AS DECIMAL(10,2))
    ELSE 0 
  END) as cod_collected,
  
  SUM(CASE 
    WHEN payment_method = 'COD' AND status IN ('created', 'dispatched', 'in_transit')
    THEN CAST(cod_amount AS DECIMAL(10,2))
    ELSE 0 
  END) as cod_pending,
  
  SUM(CASE 
    WHEN payment_method = 'COD' AND status = 'cancelled'
    THEN CAST(cod_amount AS DECIMAL(10,2))
    ELSE 0 
  END) as cod_lost

FROM daily_orders
GROUP BY report_date, payment_method
ORDER BY report_date DESC, payment_method;
