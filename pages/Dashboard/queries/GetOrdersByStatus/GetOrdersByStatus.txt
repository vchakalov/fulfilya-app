  SELECT
      status_label,
      COUNT(*) as count,
      CASE
          WHEN status_label = 'Pending' THEN '#fbbf24'
          WHEN status_label = 'In Transit' THEN '#3b82f6'
          WHEN status_label = 'Delivered' THEN '#10b981'
          WHEN status_label = 'Cancelled' THEN '#ef4444'
          ELSE '#6b7280'
      END as color
  FROM (
      SELECT
          o.uuid,
          CASE
              WHEN o.status = 'created' THEN 'Pending'
              WHEN o.status IN ('dispatched', 'in_transit', 'started') THEN 'In Transit'
              WHEN o.status IN ('delivered', 'order_delivered')
                   OR (o.status = 'completed' AND NOT EXISTS (
                       SELECT 1 FROM activity a
                       WHERE a.subject_id = o.uuid
                       AND JSON_EXTRACT(a.properties, '$.attributes.status') IN ('order_returned', 'order_cancel')
                   )) THEN 'Delivered'
              WHEN o.status IN ('canceled')
                   OR EXISTS (
                       SELECT 1 FROM activity a
                       WHERE a.subject_id = o.uuid
                       AND JSON_EXTRACT(a.properties, '$.attributes.status') IN ('order_returned', 'order_cancel')
                   ) THEN 'Cancelled'
              ELSE 'Other'
          END as status_label
      FROM orders o
      WHERE o.created_at >= DATE_FORMAT(NOW(), '%Y-%m-01')
        AND o.customer_uuid = '{{appsmith.store.customer_uuid}}'
  ) as status_data
  GROUP BY status_label
