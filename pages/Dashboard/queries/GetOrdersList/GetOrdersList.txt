SELECT
    o.public_id as order_id,
    o.internal_id as internal_reference,
    o.status,
    o.created_at as created_date,

    -- Customer information (from contacts table)
    COALESCE(c.name, 'Unknown') as customer_name,
    COALESCE(c.email, 'No Email') as customer_email,
    COALESCE(c.phone, 'No Phone') as customer_phone,

    -- Location information (from places table via payloads)
    COALESCE(pickup.name, 'Unknown Pickup') as pickup_location,
    COALESCE(dropoff.name, 'Unknown Dropoff') as delivery_address,

    -- Driver information (from users table via drivers)
    COALESCE(d_user.name, 'Unassigned') as driver_name,

    -- Tracking information
    COALESCE(tn.tracking_number, '') as tracking_number,

    -- Payment information (from custom_field_values)
    COALESCE(MAX(CASE WHEN cf.name = 'payment-method' THEN cfv.value END), 'Not Specified') as payment_method,
    COALESCE(MAX(CASE WHEN cf.name = 'amount:' THEN cfv.value END), '$0.00') as total_amount,

    -- Store information
    COALESCE(MAX(CASE WHEN cf.name = 'store-type' THEN cfv.value END), 'Unknown') as store_type,
    COALESCE(MAX(CASE WHEN cf.name = 'order-source-id' THEN cfv.value END), '') as external_order_id,

    -- Status display
  CASE
      WHEN o.status = 'order_returned' THEN 'Returned'
      WHEN o.status = 'order_cancel' THEN 'Cancelled'
      WHEN o.status = 'created' THEN 'Pending'
      WHEN o.status = 'dispatched' THEN 'In Transit'
      WHEN o.status = 'started' THEN 'In Transit'
      WHEN o.status = 'order_delivered' THEN 'Delivered'
      WHEN o.status = 'completed' AND EXISTS (
				SELECT 1 FROM activity a
				WHERE a.subject_id = o.uuid
				AND JSON_EXTRACT(a.properties, '$.attributes.status') = 'order_returned'
			) THEN 'Returned'
      WHEN o.status = 'completed' THEN 'Delivered'
      ELSE o.status
  END as status_display

  FROM orders o
  -- Customer information
  LEFT JOIN contacts c ON o.customer_uuid = c.uuid

  -- Driver information  
  LEFT JOIN drivers d ON o.driver_assigned_uuid = d.uuid
  LEFT JOIN users d_user ON d.user_uuid = d_user.uuid

  -- Location information
  LEFT JOIN payloads p ON o.payload_uuid = p.uuid
  LEFT JOIN places pickup ON p.pickup_uuid = pickup.uuid
  LEFT JOIN places dropoff ON p.dropoff_uuid = dropoff.uuid

  -- Tracking information
  LEFT JOIN tracking_numbers tn ON o.tracking_number_uuid = tn.uuid

  -- Custom field values (payment method, COD amount, etc.)
  LEFT JOIN custom_field_values cfv ON o.uuid = cfv.subject_uuid AND cfv.subject_type = 'Fleetbase\\FleetOps\\Models\\Order'
  LEFT JOIN custom_fields cf ON cfv.custom_field_uuid = cf.uuid

  WHERE o.customer_uuid = '{{appsmith.store.customer_uuid}}' and
	o.deleted_at is null

  GROUP BY o.uuid, o.public_id, o.internal_id, o.status, o.created_at, c.name, c.email, c.phone, pickup.name, dropoff.name,
  d_user.name, tn.tracking_number
  ORDER BY o.created_at DESC
  LIMIT 100;
